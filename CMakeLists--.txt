add_definitions("-Wall -g")

find_package(PkgConfig REQUIRED)
pkg_search_module(GLIB2 REQUIRED glib-2.0)
pkg_search_module(HIREDIS REQUIRED hiredis)
find_package(Boost REQUIRED COMPONENTS filesystem system)

# 设置包含目录
include_directories(${GLIB2_INCLUDE_DIRS})
# 添加pthread库
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 -pthread")

set(SRC
    common/log.h
    common/slice.h
    common/status.cpp
    common/bytebuffer.cpp
    common/random_device.cpp
    raft/proto.cpp
    raft/config.cpp
    raft/raft.cpp
    raft/storage.cpp
    raft/node.cpp
    raft/raft_log.cpp
    raft/unstable.cpp
    raft/progress.cpp
    raft/readonly.cpp
    raft/raft_status.cpp
    raft/ready.cpp
    raft/util.cpp
    server/raft_node.cpp
    server/redis_session.cpp
    server/redis_store.cpp
    snap/snapshotter.cpp
    transport/proto.h
    transport/transport.h
    transport/transport.cpp
    transport/peer.h
    transport/raft_server.cpp
    transport/peer.cpp
    wal/wal.cpp)

add_library(raft-kv++ ${SRC})
# 设置链接库
target_link_libraries(raft-kv++ ${LIBS} ${GLIB2_LIBRARIES} ${Boost_FILESYSTEM_LIBRARY} ${Boost_SYSTEM_LIBRARY} ${HIREDIS_LIBRARIES})

add_executable(raft-kv raft-kv.cpp)
target_link_libraries(raft-kv ${LIBS} raft-kv++)
